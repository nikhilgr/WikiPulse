// ----- Utilities ------------------------------------------------------
const UA = 'WikiPulse-Prototype/0.2 (https://wikipulse.live; nikhil@amplitude.com)';
const IS_REAL_ARTICLE = (a) => a && !a.startsWith('Special:') && !a.startsWith('Wikipedia:') && !a.startsWith('Portal:') && !a.startsWith('File:') && !a.startsWith('Category:') && !a.startsWith('Help:') && !a.startsWith('Talk:') && a !== 'Main_Page' && a !== '-';
const SHORT = (n) => n >= 1e6 ? (n/1e6).toFixed(1)+'M' : n >= 1e3 ? (n/1e3).toFixed(1)+'k' : String(n);
const pad = (n) => String(n).padStart(2,'0');

function getReportDate() {
  const d = new Date(Date.now() - 36 * 3600 * 1000);
  return { y: d.getUTCFullYear(), m: pad(d.getUTCMonth()+1), dd: pad(d.getUTCDate()), d };
}

// ----- API calls ------------------------------------------------------
async function fetchGlobalTop() {
  const r = getReportDate();
  const url = `https://wikimedia.org/api/rest_v1/metrics/pageviews/top/en.wikipedia/all-access/${r.y}/${r.m}/${r.dd}`;
  const res = await fetch(url);
  if (!res.ok) throw new Error('global top API failed');
  const data = await res.json();
  return data.items[0].articles;
}

async function fetchCountryTop(code) {
  const r = getReportDate();
  const url = `https://wikimedia.org/api/rest_v1/metrics/pageviews/top-per-country/${code}/all-access/${r.y}/${r.m}/${r.dd}`;
  try {
    const res = await fetch(url);
    if (!res.ok) return null;
    const data = await res.json();
    const articles = (data.items?.[0]?.articles || [])
      .filter(a => IS_REAL_ARTICLE(a.article))
      .filter(a => !a.project || a.project.startsWith('en'))
      .map(a => ({ article: a.article, views: a.views_ceil || a.views || 0, rank: a.rank }))
      .slice(0, 10);
    return articles;
  } catch (e) { return null; }
}

async function fetchSummary(title) {
  try {
    const url = `https://en.wikipedia.org/api/rest_v1/page/summary/${encodeURIComponent(title)}?redirect=true`;
    const res = await fetch(url);
    if (!res.ok) return null;
    return await res.json();
  } catch (e) { return null; }
}

// ----- Category guesser (for kickers) ---------------------------------
function categoryOf(desc) {
  if (!desc) return 'Topic';
  const d = desc.toLowerCase();
  if (/(film|movie|actor|actress|director|series|tv |television)/.test(d)) return 'Screen';
  if (/(politician|president|prime minister|senator|election|party)/.test(d)) return 'Politics';
  if (/(footballer|cricket|tennis|nba|nfl|athlete|sport|olymp)/.test(d)) return 'Sport';
  if (/(singer|musician|band|rapper|album|song)/.test(d)) return 'Music';
  if (/(scientist|physicist|chemist|biologist|engineer)/.test(d)) return 'Science';
  if (/(author|writer|novelist|poet|book)/.test(d)) return 'Letters';
  if (/(company|brand|business|ceo)/.test(d)) return 'Business';
  if (/(country|city|capital|region|island)/.test(d)) return 'Places';
  if (/(war|battle|military|treaty|empire)/.test(d)) return 'History';
  return 'Topic';
}

// ----- Cards ----------------------------------------------------------
async function renderCards(articles, contextLabel) {
  const grid = document.getElementById('articles');
  grid.classList.add('loading');
  grid.innerHTML = '';
  if (!articles || articles.length === 0) {
    grid.classList.remove('loading');
    grid.innerHTML = `<div style="grid-column: span 4; text-align: center; padding: 60px 20px; color: var(--muted); font-family: 'Fraunces', serif; font-style: italic; font-size: 18px;">No data available for this selection. Wikipedia may not surface country-level pageviews for regions with low traffic, or the day's data hasn't published yet.</div>`;
    return;
  }
  document.getElementById('gridContext').textContent = contextLabel;

  // First pass: render skeletons with what we know
  articles.forEach((a, i) => {
    const card = document.createElement('div');
    card.className = 'card' + (i === 0 ? ' feature' : '');
    card.dataset.article = a.article;
    card.innerHTML = `
      <div class="img placeholder"><span>loading…</span><div class="rank">№ ${a.rank}</div></div>
      <div class="body">
        <div class="kicker">Topic</div>
        <div class="title">${a.article.replace(/_/g, ' ')}</div>
        <div class="desc"></div>
        <div class="views"><b>${SHORT(a.views)}</b> &nbsp;views in 24h</div>
      </div>`;
    card.onclick = () => openArticle(a.article);
    grid.appendChild(card);
  });

  // Second pass: enrich with summaries
  await Promise.all(articles.map(async (a) => {
    let summary = state.summaryCache.get(a.article);
    if (!summary) {
      summary = await fetchSummary(a.article);
      if (summary) state.summaryCache.set(a.article, summary);
    }
    if (!summary) return;
    const card = grid.querySelector(`[data-article="${CSS.escape(a.article)}"]`);
    if (!card) return;
    const thumb = summary.thumbnail || summary.originalimage;
    const imgDiv = card.querySelector('.img');
    if (thumb) {
      imgDiv.className = 'img';
      imgDiv.innerHTML = `<div class="rank">№ ${a.rank}</div>`;
      imgDiv.style.backgroundImage = `url('${thumb.source}')`;
    } else {
      imgDiv.innerHTML = `<span>no image</span><div class="rank">№ ${a.rank}</div>`;
    }
    const title = (summary.displaytitle || a.article).replace(/<[^>]+>/g, '');
    card.querySelector('.title').textContent = title;
    card.querySelector('.kicker').textContent = categoryOf(summary.description);
    const isFeature = card.classList.contains('feature');
    const desc = isFeature ? (summary.extract || summary.description || '').slice(0, 240) : (summary.description || '');
    card.querySelector('.desc').textContent = desc;
  }));

  grid.classList.remove('loading');
}

// ----- Detail panel ---------------------------------------------------
async function openArticle(title) {
  const panel = document.getElementById('panel');
  panel.classList.add('open');
  document.getElementById('panelTitle').textContent = title.replace(/_/g, ' ');
  document.getElementById('panelDesc').textContent = 'Loading…';
  document.getElementById('panelExtract').textContent = '';
  document.getElementById('panelImg').style.backgroundImage = '';
  document.getElementById('panelContext').textContent = state.selected ? countryName(state.selected) : 'Global';

  const articles = state.selected
    ? state.countryCache.get(state.selected)
    : state.globalTop;
  const article = articles?.find(a => a.article === title);
  document.getElementById('panelRank').textContent = article ? '#' + article.rank : '—';
  document.getElementById('panelViews').textContent = article ? SHORT(article.views) : '—';

  let summary = state.summaryCache.get(title);
  if (!summary) {
    summary = await fetchSummary(title);
    if (summary) state.summaryCache.set(title, summary);
  }
  if (summary) {
    const thumb = summary.originalimage || summary.thumbnail;
    if (thumb) document.getElementById('panelImg').style.backgroundImage = `url('${thumb.source}')`;
    document.getElementById('panelTitle').innerHTML = summary.displaytitle || title.replace(/_/g, ' ');
    document.getElementById('panelDesc').textContent = summary.description || '';
    document.getElementById('panelExtract').textContent = summary.extract || '';
    document.getElementById('panelLink').href = summary.content_urls?.desktop?.page || `https://en.wikipedia.org/wiki/${title}`;
    document.getElementById('panelKicker').textContent = categoryOf(summary.description) + (state.selected ? ' · ' + countryName(state.selected).toUpperCase() : ' · GLOBAL');
  }
}

function closePanel() { document.getElementById('panel').classList.remove('open'); }

// ----- View chip + reset ----------------------------------------------
function countryName(a2) {
  // try numeric lookup → name from topojson properties
  for (const [num, code] of Object.entries(NUM_TO_A2)) {
    if (code === a2) {
      const n = state.countryNames.get(num);
      if (n) return n;
    }
  }
  return a2;
}

function setViewGlobal() {
  state.selected = null;
  const chip = document.getElementById('viewChip');
  chip.classList.add('global');
  document.getElementById('viewName').textContent = 'Global · the whole world';
  document.getElementById('gridContext').textContent = 'globally';
  d3.selectAll('path.country').classed('selected', false);
}

function setViewCountry(a2) {
  state.selected = a2;
  const chip = document.getElementById('viewChip');
  chip.classList.remove('global');
  document.getElementById('viewName').textContent = countryName(a2);
}

function resetView() {
  setViewGlobal();
  renderCards(state.globalTop, 'globally');
}

  if (!a2) {
    toast('No country code mapping for this region.');
    return;
  }
  setViewCountry(a2);
  d3.selectAll('path.country').classed('selected', false);
  if (numericId) d3.select(`path.country[data-num="${numericId}"]`).classed('selected', true);

  const name = countryName(a2);
  toast(`Loading what ${name} is reading…`);

  let articles = state.countryCache.get(a2);
  if (!articles) {
    articles = await fetchCountryTop(a2);
    if (articles) state.countryCache.set(a2, articles);
  }
  if (!articles || articles.length === 0) {
    renderCards(null, `in ${name}`);
    toast(`Wikipedia doesn't surface pageview data for ${name}.`);
    return;
  }
  renderCards(articles, `in ${name}`);
}

// ----- Region filter --------------------------------------------------
function setRegion(region) {
  state.region = region;
  document.querySelectorAll('.pill').forEach(p => p.classList.toggle('active', p.dataset.region === region));
  d3.selectAll('path.country').classed('dim', function() {
    if (region === 'ALL') return false;
    const a2 = NUM_TO_A2[this.dataset.num];
    if (!a2) return true;
    return A2_TO_REGION[a2] !== region;
  });
}

// ----- Map rendering --------------------------------------------------
let topoCache = null;

async function renderMap() {
  if (!topoCache) {
    const url = 'https://cdn.jsdelivr.net/npm/world-atlas@2/countries-110m.json';
    topoCache = await fetch(url).then(r => {
      if (!r.ok) throw new Error('topojson load failed');
      return r.json();
    });
  }
  const topo = topoCache;
  const countries = topojson.feature(topo, topo.objects.countries).features;

  // Cache country names by numeric ID
  countries.forEach(c => state.countryNames.set(String(c.id).padStart(3,'0'), c.properties.name));

  document.getElementById('statCountries').textContent = countries.length;

  const wrap = document.getElementById('mapStage');
  const svg = d3.select('#world');
  const width = wrap.clientWidth;
  const height = wrap.clientHeight;
  svg.attr('viewBox', `0 0 ${width} ${height}`);
  svg.selectAll('*').remove();

  // Fit the projection then shift it right so it sits between the two overlays
  // (headline on left ~520px, controls on right ~360px). The visible "map" area
  // is centered between them, not on the canvas.
  const projection = d3.geoNaturalEarth1().fitSize([width * 0.92, height * 0.96], { type: 'Sphere' });
  // Push origin ~6% right so North America isn't trapped behind the headline panel
  const isWide = width > 900;
  projection.translate([width * (isWide ? 0.56 : 0.5), height / 2]);
  const path = d3.geoPath(projection);

  // Background sphere (subtle outline)
  svg.append('path').attr('class', 'outline').attr('d', path({ type: 'Sphere' }));

  // Graticule
  const graticule = d3.geoGraticule10();
  svg.append('path').attr('class', 'graticule').attr('d', path(graticule));

  // Countries
  const tip = document.getElementById('mapTip');
  svg.selectAll('path.country')
    .data(countries)
    .enter()
    .append('path')
    .attr('class', d => {
      const a2 = NUM_TO_A2[String(d.id).padStart(3,'0')];
      return 'country' + (PRELOAD.includes(a2) ? ' has-data' : '');
    })
    .attr('d', path)
    .attr('data-num', d => String(d.id).padStart(3,'0'))
    .on('mousemove', function(ev, d) {
      const num = String(d.id).padStart(3,'0');
      const a2 = NUM_TO_A2[num];
      const cached = a2 ? state.countryCache.get(a2) : null;
      let topHtml = '';
      if (cached && cached.length > 0) {
        topHtml = `<div class="top"><span class="label">Top three articles</span>${cached.slice(0,3).map((a,i) => `${i+1}. ${a.article.replace(/_/g,' ')}`).join('<br>')}</div>`;
      } else {
        topHtml = `<div class="top" style="font-style:italic;color:var(--muted);">Click to load what ${d.properties.name} is reading.</div>`;
      }
      tip.innerHTML = `<div class="country">${d.properties.name}</div>${topHtml}<div class="hint">${cached ? 'Click to focus' : '↗ Click to load'}</div>`;
      const rect = wrap.getBoundingClientRect();
      let left = ev.clientX - rect.left + 16;
      let top = ev.clientY - rect.top - 20;
      if (left + 300 > rect.width) left = ev.clientX - rect.left - 316;
      if (top < 16) top = ev.clientY - rect.top + 24;
      tip.style.left = left + 'px';
      tip.style.top = top + 'px';
      tip.style.opacity = 1;
    })
    .on('mouseleave', function() { tip.style.opacity = 0; })
    .on('click', function(ev, d) {
      const num = String(d.id).padStart(3,'0');
      const a2 = NUM_TO_A2[num];
      selectCountry(a2, num);
    });

  // Pre-load tooltips for top countries (so hover preview shows top 3)
  toast('Pre-loading data for top countries…');
  await chunkedFetch(PRELOAD, async (code) => {
    if (state.countryCache.has(code)) return;
    const articles = await fetchCountryTop(code);
    if (articles) state.countryCache.set(code, articles);
  }, 4);
  toast('Map ready. Click any country to dive in.');
}

async function chunkedFetch(items, fn, chunkSize = 5) {
  for (let i = 0; i < items.length; i += chunkSize) {
    await Promise.all(items.slice(i, i + chunkSize).map(fn));
  }
}

// ----- Init -----------------------------------------------------------
function toast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  clearTimeout(toast._t);
  toast._t = setTimeout(() => t.classList.remove('show'), 2800);
}

function setDateLine() {
  const now = new Date();
  const opts = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
  document.getElementById('dateLine').textContent = now.toLocaleDateString('en-US', opts).toUpperCase();
  const yyyymmdd = now.toISOString().slice(0,10).replace(/-/g,'').slice(2);
  document.getElementById('editionNum').textContent = yyyymmdd;
}

async function boot() {
  setDateLine();
  document.getElementById('mapLoading').classList.add('show');

  // Wire up region pills
  document.querySelectorAll('.pill').forEach(p => p.addEventListener('click', () => setRegion(p.dataset.region)));

  // Parallel: load global top + map
  try {
    const [globalTop] = await Promise.all([fetchGlobalTop()]);
    const articles = globalTop.filter(a => IS_REAL_ARTICLE(a.article)).slice(0, 50);
    state.globalTop = articles;
    document.getElementById('statTotal').textContent = SHORT(articles.reduce((s,a)=>s+a.views, 0));
    renderCards(articles, 'globally');
  } catch (e) {
    console.error(e);
    toast('Could not reach Wikipedia. Check your network?');
  }

  try {
    await renderMap();
  } catch (e) {
    console.error(e);
    toast('Map data could not load. Network blocked?');
  }
  document.getElementById('mapLoading').classList.remove('show');
}

  }, 250);
